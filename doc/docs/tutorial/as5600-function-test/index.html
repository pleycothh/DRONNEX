AS5600 Function testâ€‹

Sbus-streamer.ino
#include <Wire.h>

#define AS5600_ADDR 0x36
#define SDA_PIN 21
#define SCL_PIN 20

void setup() {
  Serial.begin(115200);
  Wire.begin(SDA_PIN, SCL_PIN);

  delay(500);
  Serial.println("ğŸ” Starting AS5600 I2C Scan...");

  Wire.beginTransmission(AS5600_ADDR);
  byte error = Wire.endTransmission();
  if (error == 0) {
    Serial.println("âœ… AS5600 found at 0x36");
  } else {
    Serial.println("âŒ AS5600 not detected");

uint16_t readAS5600RawAngle() {
  Wire.beginTransmission(AS5600_ADDR);
  Wire.write(0x0C); // Start at raw angle MSB
  Wire.endTransmission();

  Wire.requestFrom(AS5600_ADDR, 2);
  uint8_t msb = Wire.read();
  uint8_t lsb = Wire.read();

  return ((msb << 8) | lsb) & 0x0FFF; // 12-bit angle
}
